\begin{frame}
\frametitle{Present Value}
  \begin{block}{Pricing a future cashflow}
    \begin{equation}
      FV = N (1 + r) \Leftrightarrow N = \frac{FV}{1+r}
    \end{equation}
    \begin{itemize}
      \item $FV$ : Future Value
      \item $N$  : Amount invested
      \item $r$  : Interest Rate
    \end{itemize}
  \end{block}
\end{frame}
